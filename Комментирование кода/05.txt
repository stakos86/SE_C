/*
Задача 5.
Найдите количество чётных цифр данного натурального числа.
*/

#include <iostream> // Подключаем библиотеку iostream для ввода/вывода
using namespace std; // Используем пространство имен std, чтобы не писать std:: перед cout и cin

int main() // Главная функция программы
{
    setlocale(LC_ALL, ""); // Устанавливаем локаль для корректного отображения символов на русском языке
    int number = 0, counter = 0; // Объявляем и инициализируем две переменные: number - число, которое мы будем проверять, и counter - счетчик четных цифр
    cin >> number; // Считываем число с консоли
    while (number) // Цикл while продолжается, пока number не станет равным нулю
    {
        if ((number % 10) % 2 == 0) // Если последняя цифра числа (полученная с помощью number % 10) делится на 2 без остатка
            counter++; // Увеличиваем счетчик на 1
        number /= 10; // Удаляем последнюю цифру числа (делим на 10)
    }

    cout << "Количество чётных цифр: " << counter; // Выводим количество найденных четных цифр
}
